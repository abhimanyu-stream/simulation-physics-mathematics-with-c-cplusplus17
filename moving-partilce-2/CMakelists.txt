

cmake_minimum_required(VERSION 3.10)

# Project name
project(ParticleMoving-2)
set(CMAKE_VERBOSE_MAKEFILE ON)
add_definitions(-DSFML_STATIC)
# Set the C++ standard
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED True)

# Add the executable
add_executable(Fish-Simulation-Water-SFML Fish-Simulation-Water-SFML.cpp)

# SFML path setup
set(SFML_DIR "C:/SFML-2.6.2/lib/cmake/SFML")

# Add the SFML lib directory to the linker path
link_directories("C:/SFML-2.6.2/lib")

# Find SFML package
find_package(SFML 2.6 COMPONENTS graphics window system REQUIRED)

# Include SFML headers and link libraries
target_include_directories(Fish-Simulation-Water-SFML PRIVATE "C:/SFML-2.6.2/include")
target_link_libraries(Fish-Simulation-Water-SFML PRIVATE sfml-graphics sfml-window sfml-system)
